\begin{algorithm}
\SetKwFunction{Canny}{canny}
\SetKwProg{Fn}{def}{:}{end}

\Fn{\Canny{image}}{
    \nonl \textbf{Convert to Grayscale}\;
    gray = cv2.cvtColor(image, cv2.COLOR\_BGR2GRAY)\;
    
    \nonl \textbf{Apply Gaussian Blur}\;
    blur = cv2.GaussianBlur(gray, (5, 5), 0)\;
    
    \nonl \textbf{Apply Canny Edge Detection}\;
    canny = cv2.Canny(blur, 50, 175)\;
    \textbf{Return Result}\;
    return canny\;
}

\caption{Canny Function}
\label{algo_canny}
\end{algorithm}
